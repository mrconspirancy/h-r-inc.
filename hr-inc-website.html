Backend - API Endpoint Structure

// routes/services.js (Node.js + Express)
class ServiceController {
  async getAllServices(req, res) {
    const services = await Service.find();
    res.json(services);
  }
  
  async createConsultation(req, res) {
    const consultation = new Consultation(req.body);
    await consultation.save();
    res.status(201).json(consultation);
  }
}

module.exports = new ServiceController();

Database - Service Model

// models/Service.js (Mongoose ORM)
const serviceSchema = new Schema({
  title: { type: String, required: true },
  category: { 
    type: String, 
    enum: ['defence', 'cyber', 'geopolitics', 'health']
  },
  description: String,
  caseStudies: [{ type: Schema.Types.ObjectId, ref: 'CaseStudy' }],
  createdAt: { type: Date, default: Date.now }
});

module.exports = mongoose.model('Service', serviceSchema);
